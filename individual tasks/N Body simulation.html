def getAcc(pos, mass, G, softening):
    x = pos[:, 0:1]
    y = pos[:, 1:2]

    dx = x.T - x
    dy = y.T - y

    inv_r3 = (dx ** 2 + dy ** 2 + softening ** 2) ** (-1.5)

    ax = G * (dx * inv_r3) @ mass
    ay = G * (dy * inv_r3) @ mass

    a = np.hstack((ax, ay))

    return a
